CREATE TABLE orders (

SELECT
    product_name,
    unit_price,
    CASE
        WHEN unit_price < 1000                      THEN 'Budget'
        WHEN unit_price BETWEEN 1000 AND 3000       THEN 'Mid-Range'
        WHEN unit_price > 3000                      THEN 'Premium'
    END AS price_tier
